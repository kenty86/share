	void dijkstra(struct element *a[], int n, int s)
	{
		int i, u, v, sizeofT;
		struct element *l;
	
		for (i = 1; i <= n; i++){
			T[i] = 1;
			D[i] = INFTY; //2n回分初期化
		}
		T[s] = 0;
		D[s] = 0;
		for (l = a[s]->next; l != NULL; l = l->next)
			D[l->vertex] = l->weight;		//nlog(size)回計算
		sizeofT = n-1;
		while (sizeofT > 0){
			for (v = 1; v <= n; v++) 
				if (T[v] == 1)
		printf("v[%d], ", v);
				else
		printf("      ");
			for (v = 2; v <= n; v++) 
				printf("D[%d]=%d, ", v, D[v]);
			printf("\n");
			u = findmin(T, n);  //n回探索をする。
			T[u] = 0;          
			sizeofT--;
			for (v = 1; v <= n; v++)
				if (T[v] == 1)
		D[v] = min(D[v], D[u]+w[u][v]);//n回演算する。
		//2n+n+n=2nよって時間計算量O(2n)
		}
	}
